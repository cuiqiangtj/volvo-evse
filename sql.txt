-- CREATE DATABASE `volvo`  DEFAULT CHARACTER SET utf8 ;

-- volvo.connector definition

CREATE TABLE `connector` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `EVSE_ID` int(11) DEFAULT NULL,
  `STANDARD` varchar(100) DEFAULT NULL COMMENT '标准',
  `POWER_LEVEL` varchar(100) DEFAULT NULL COMMENT '功率',
  `VOLTAGE` varchar(100) DEFAULT NULL COMMENT '电压',
  `CREATE_TIME` timestamp NULL DEFAULT NULL,
  `UPDATE_TIME` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=7 DEFAULT CHARSET=utf8 COMMENT='EVSE的充电连接器';


-- volvo.evse definition

CREATE TABLE `evse` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `LOCATION_ID` int(11) DEFAULT NULL,
  `EVSE_CODE` varchar(100) DEFAULT NULL COMMENT '唯一标识',
  `STATUS` varchar(20) DEFAULT NULL COMMENT '运行状态,INITIAL,AVAILABLE,BLOCKED,INOPERATIVE,REMOVED',
  `CREATE_TIME` timestamp NULL DEFAULT NULL,
  `UPDATE_TIME` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8 COMMENT='EVSE (Electric Vehicle Supply Equipment)';


-- volvo.location definition

CREATE TABLE `location` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `NAME` varchar(100) DEFAULT NULL COMMENT '名称',
  `ADDRESS` varchar(100) DEFAULT NULL COMMENT '地址',
  `COORDINATES` varchar(100) DEFAULT NULL COMMENT '坐标',
  `BUSINESS_HOURS` varchar(100) DEFAULT NULL COMMENT '营业时间',
  `CREATE_TIME` timestamp NULL DEFAULT NULL,
  `UPDATE_TIME` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=17 DEFAULT CHARSET=utf8 COMMENT='充电站点';
